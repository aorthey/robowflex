51 std::shared_ptr<Pool::Job<planning_interface::MotionPlanResponse>>
52 PoolPlanner::submit(const SceneConstPtr &scene, const planning_interface::MotionPlanRequest &request)
53 {
54  return pool_.submit(make_function([&] {
55  std::unique_lock<std::mutex> lock(mutex_);
56  cv_.wait(lock, [&] { return !planners_.empty(); });
57 
58  auto planner = planners_.front();
59  planners_.pop();
60 
61  lock.unlock();
62 
63  auto result = planner->plan(scene, request);
64 
65  lock.lock();
66  planners_.emplace(planner);
67  cv_.notify_one();
68 
69  return result;
70  }));
71 }

107 bool OMPL::loadOMPLConfig(IO::Handler &handler, const std::string &config_file,
108  std::vector<std::string> &configs)
109 {
110  if (config_file.empty())
111  return false;
112 
113  auto &config = IO::loadFileToYAML(config_file);
114  if (!config.first)
115  {
116  RBX_ERROR("Failed to load planner configs.");
117  return false;
118  }
119 
120  handler.loadYAMLtoROS(config.second);
121 
122  auto &planner_configs = config.second["planner_configs"];
123  if (planner_configs)
124  {
125  for (YAML::const_iterator it = planner_configs.begin(); it != planner_configs.end(); ++it)
126  configs.push_back(it->first.as<std::string>());
127  }
128 
129  return true;
130 }
